If a function $f(t)$ is given by its Fourier series i.e. 
\begin{eqnarray*}
f(t) &=& \frac{a_0}{2} + \sum_{k=1}^{\infty} [a_k\cos(k\omega t) + b_k\sin(k\omega t)]
\end{eqnarray*}
The RMS value $F_{\mathrm{rms}}$ of $f(t)$ is  
\begin{eqnarray*}
F_{\mathrm{rms}} &=& \frac{{a_0}^2}{4} + \frac{1}{2} \sum_{k=1}^{\infty} {a_k}^2 + {b_k}^2
\end{eqnarray*}

$\textbf{Proof:}$

The RMS value of a function $f(t)$ is, by definition, given by 
\begin{eqnarray*} 
F_{\mathrm{rms}} &=& \frac{1}{T} \int_{t_0}^{t_0+T} [f(t)]^2 dt
\end{eqnarray*}
Then, 
\begin{align*}
F_{\mathrm{rms}} &= \frac{1}{T} \int_{t_0}^{t_0+T} [f(t)]^2 dt \\
&= \frac{1}{T}\int_{t_0}^{t_0+T} f(t)\cdot f(t) dt \\
&= \frac{1}{T}\int_{t_0}^{t_0+T} f(t)\cdot \Big( \frac{a_0}{2} + \sum_{k=1}^{\infty}\Big[a_k\cos(k\omega t) + b_k\sin(k\omega t)\Big] \Big) dt \\
&= \frac{1}{T}\int_{t_0}^{t_0+T}\Big( \frac{a_0 f(t)}{2} + \sum_{k=1}^{\infty}\Big[a_k f(t)\cos(k\omega t) + b_k f(t)\sin(k\omega t)\Big] \Big) dt \\
&= \frac{1}{T}\Big( \int_{t_0}^{t_0+T} \frac{a_0 f(t)}{2}dt + \sum_{k=1}^{\infty} \Big[\int_{t_0}^{t_0+T} a_k f(t)\cos(k\omega t) dt + \int_{t_0}^{t_0+T} b_k f(t)\sin(k\omega t) dt \Big] \Big) \\
&= \frac{1}{T}\Big( \frac{a_0}{2}\int_{t_0}^{t_0+T} f(t) dt + \sum_{k=1}^{\infty} \Big[a_k\int_{t_0}^{t_0+T} f(t)\cos(k\omega t) dt + b_k\int_{t_0}^{t_0+T}f(t)\sin(k\omega t) dt \Big] \Big) \\
&= \frac{1}{T}\Big( \frac{a_0}{2}(a_0 \frac{T}{2}) + \sum_{k=1}^{\infty} \Big[a_k( a_k\frac{T}{2}) + b_k(b_k\frac{T}{2}) \Big] \Big) \\
&= \frac{{a_0}^2}{4} + \sum_{k=1}^{\infty}\frac{{a_k}^2}{2} + \frac{{b_k}^2}{2} \\
&= \frac{{a_0}^2}{4} + \frac{1}{2}\sum_{k=1}^{\infty}{a_k}^2 + {b_k}^2
\end{align*}
