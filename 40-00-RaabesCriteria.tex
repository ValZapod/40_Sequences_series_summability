\begin{thmplain}
\,The series\, $a_1\!+\!a_2\!+\!a_3\!+\cdots$\, with positive \PMlinkescapetext{terms} is
\begin{itemize}
 \item convergent if, starting from some value of $n$, its \PMlinkescapetext{terms} fulfil the condition
   $$\frac{a_{n+1}}{a_n} \leqq 1-\frac{\mu}{n}$$
where $\mu$ is a \PMlinkescapetext{constant} and $> 1$;
 \item divergent if, starting from some value of $n$, its \PMlinkescapetext{terms} fulfil the condition
   $$\frac{a_{n+1}}{a_n} \geqq 1-\frac{1}{n}-\frac{M}{n^2}$$
where $M$ is a \PMlinkescapetext{constant}.
\end{itemize}
\end{thmplain}
