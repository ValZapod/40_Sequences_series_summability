$\textbf{Some Unconventional Syntax:}$
\begin{eqnarray*}
A^{x}\hat{e}_x + A^{y}\hat{e}_y + A^{z}\hat{e}_z &=& A_{x}\hat{e}_x + A_{y}\hat{e}_y + A_{\phi}\hat{e}_z \\
dx\wedge dy &=& dx dy \\ 
dx\wedge dy\wedge dz &=& dx dy dz \\
\int_{t_1}^{t_2} dt f(t) g(t) &=& \int_{t_1}^{t_2} f(t)g(t) dt \\ 
{\partial}_x &=& \frac{\partial}{\partial x} \\
d\Sigma &=& \mbox{material surface element} \\
f(\vec{r}) &=& f(x,y,z) \\
\mathbb{E}^3 &=& \mathbb{R}^3\\
\end{eqnarray*}

$\textbf{Definitions:}$
\begin{eqnarray*}
\vec{A}\circ\vec{B} &\equiv & \sum_{i=1}^3 A_i B_i \\
\end{eqnarray*}

\begin{eqnarray*}
\vec{A}\times\vec{B} &\equiv & \left|\begin{array}{ccc}
\bold{\hat{e}_x} & \bold{\hat{e}_y } & \bold{\hat{e}_z}\\
A_x & A_y & A_z \\
B_x & B_y & B_z
\end{array}\right|
\end{eqnarray*}

\begin{eqnarray*}
[\vec{A},\vec{B},\vec{C}] &\equiv & \vec{A}\circ(\vec{B}\times\vec{C}) =  (\vec{A}\times\vec{B})\circ\vec{C} = \Bigg|\begin{array}{ccc}
A_x & A_y & A_z \\
B_x & B_y & B_z \\
C_x & C_y & C_z
\end{array}\Bigg| 
\end{eqnarray*}

\begin{eqnarray*}
\Delta &\equiv& {\nabla}^2 \quad \mbox{(Laplace operator or Laplacian)}\\
{\Box}^2 &\equiv& {\nabla}^2 - \frac{1}{c^2}{{\partial}_t}^2 \quad \mbox{(D'Alembert or wave operator or D'Alembertian)}\\
{\nabla}^2\vec{A} \equiv \nabla(\nabla\circ\vec{A}) - \nabla\times(\nabla\times\vec{A}) &=& ( {\nabla}^2 A_x, {\nabla}^2 A_y, {\nabla}^2 A_z ) = \frac{\partial\vec{A}}{\partial x} + \frac{\partial\vec{A}}{\partial y} + \frac{\partial\vec{A}}{\partial z} \quad \mbox{(Vector Laplacian)}\\
A\circ\nabla &\equiv& (a_1\frac{\partial}{\partial x_1} + a_2\frac{\partial}{\partial x_2} + \ldots + a_n \frac{\partial}{\partial x_n}) \\
\end{eqnarray*}

\begin{eqnarray*}
\nabla\circ\vec{F} &=& 0 \Rightarrow \vec{F} \mbox{ is incompressible or solenoidal or divergence-free} \\
\nabla\times\vec{F} &=& 0 \Rightarrow \vec{F} \mbox{ is irrotational or conservative or curl-free} \\
laplacian(\vec{F}) &\equiv& div(grad(\vec{F}))\\
D_{\vec{u}} f &\equiv& \nabla f\circ\hat{u} \qquad \mbox{(Directional Derivative)}\\
rot_{\vec{u}}\vec{F} &\equiv& curl(\vec{F})\cdot\hat{u} \qquad \mbox{(Rotational Derivative)}\\
helicity(\vec{F}) &\equiv& curl(\vec{F})\circ\vec{F}\\
\end{eqnarray*}


\begin{eqnarray*}
\nabla\circ\vec{F} &=& \lim_{\delta V \to 0} \frac{1}{\delta V} \oiint_{\delta S}\vec{F}\circ\hat{n}\;dS\\
(\nabla\times\vec{F})\circ\vec{n} &=& \lim_{\delta S \to 0} \frac{1}{\delta S} \oint_{\delta C}\vec{F}\circ\hat{t}\;ds\\
\nabla\times\vec{F} &=& \lim_{\delta V \to 0} \frac{1}{\delta V} \oiint_{\delta S}\hat{n}\times\vec{F}\;dS\\
\nabla f &=& \lim_{\delta V \to 0} \frac{1}{\delta V} \oiint_{\delta S}\hat{n} f\;dS\\
\nabla &\equiv& \sum_i \frac{\hat{e}_i}{h_i}\frac{\partial}{\partial v_i}\\
\nabla\circ &\equiv& \sum_i \frac{\hat{e}_i}{h_i}\circ\frac{\partial}{\partial v_i}\\
\nabla\times &\equiv& \sum_i \frac{\hat{e}_i}{h_i}\times\frac{\partial}{\partial v_i}\\
\end{eqnarray*}

$\textbf{Gradient, Divergence and Curl in Curvilinear Coordinates}$
\begin{eqnarray*}
&& \nabla f = \frac{1}{h_1}\frac{\partial f}{\partial u_1}\hat{e}_1 + \frac{1}{h_2}\frac{\partial f}{\partial u_2}\hat{e}_2 + \frac{1}{h_3}\frac{\partial f}{\partial u_3}\hat{e}_3 \\
&& \nabla\circ\vec{F} = \frac{1}{h_1 h_2 h_3} \bigg[ \frac{\partial}{\partial u_1} (h_2 h_3 F_1) + \frac{\partial}{\partial u_2} (h_1 h_3 F_2) + \frac{\partial}{\partial u_3} (h_1 h_2 F_3) \bigg] \\
&& {\nabla}^2 f = \frac{1}{h_1 h_2 h_3} \bigg[ \frac{\partial}{\partial u_1}\bigg(\frac{h_2 h_3}{h_1}\frac{\partial f}{\partial u_1}\bigg) + \frac{\partial}{\partial u_2}\bigg(\frac{h_1 h_3}{h_2}\frac{\partial f}{\partial u_2}\bigg) + \frac{\partial}{\partial u_3}\bigg(\frac{h_1 h_2}{h_3}\frac{\partial f}{\partial u_3}\bigg) \bigg] \\
&& \nabla\times\vec{F} = \frac{1}{h_1 h_2 h_3}\left|\begin{array}{ccc}
h_1\hat{e}_1 & h_2\hat{e}_2 & h_3\hat{e}_3 \\
\frac{\partial}{\partial u_1} & \frac{\partial}{\partial u_2} & \frac{\partial}{\partial u_3} \\
h_1 F_1 & h_2 F_2 & h_3 F_3 
\end{array}\right| \\
&& \text{ For Cartesian   coordinates: } h_x = h_y = h_z = 1 \\
&& \text{ For Cylindrical coordinates: } h_r = 1, h_{\phi} = r, h_z = 1 \\
&& \text{ For Spherical   coordinates: } h_r = 1, h_{\theta} = r, h_{\phi} = r\sin(\theta) \\
\end{eqnarray*}

\begin{eqnarray*}
\nabla f &=& \sum_i \frac{1}{h_i} \frac{\partial f}{\partial q_i} \hat e_i \\
\nabla\times\vec{F} &=& \frac{1}{\Omega} \sum_i  \hat e_i \sum_{jk} \epsilon_{ijk} h_i \frac{\partial(h_k F_k)}{\partial q_j} \\
\nabla\cdot\vec{F} &=& \sum_i \frac{1}{\Omega} \frac{\partial}{\partial q_i} \Big( \frac{\Omega F_i}{h_i} \Big) \\
\nabla^2 f &=& \frac{1}{\Omega} \sum_i \frac{\partial}{\partial q_i} \Big(\frac{\Omega}{h_i^2} \frac{\partial f}{\partial q_i}\Big) \\
\text{  where  } \Omega \equiv \Pi h_i \\
\end{eqnarray*}

$\textbf{Inequalities:}$
\begin{eqnarray*}
&& \parallel\vec{u} + \vec{v}\parallel \; \leq \; \parallel\vec{u}\parallel + \parallel\vec{v}\parallel \qquad \mbox{Triangle Inequality} \\
&& |\vec{u}\circ\vec{v}| \; \leq \; \parallel\vec{u}\parallel \parallel\vec{v}\parallel \qquad \mbox{Cauchy-Schwarz Inequality}
\end{eqnarray*}

$\textbf{Product identities:}$
\begin{eqnarray*}
\vec{A}\circ(\vec{B}\times\vec{C}) &=& \vec{B}\circ(\vec{C}\times\vec{A}) = \vec{C}\circ(\vec{A}\times\vec{B}) \qquad \mbox{Scalar Triple Product}\\
\vec{A}\times(\vec{B}\times\vec{C}) &=& \vec{B}(\vec{A}\circ\vec{C}) - \vec{C}(\vec{A}\circ\vec{B}) \qquad \mbox{Vector Triple Product}\\
\vec{A}\times(\vec{B}\times\vec{C}) + \vec{B}\times(\vec{C}\times\vec{A}) + \vec{C}\times(\vec{A}\times\vec{B}) &=& 0 \qquad \mbox{Jacobi Identity}\\
\end{eqnarray*}

\begin{eqnarray*}
(\vec{A}\times\vec{B})\circ(\vec{C}\times\vec{D}) &=& \vec{A}\circ[\vec{B}\times(\vec{C}\times\vec{D})] = (\vec{A}\circ\vec{C})(\vec{B}\circ\vec{D}) - (\vec{B}\circ\vec{C})(\vec{A}\circ\vec{D}) \quad \mbox{Scalar Quadruple Product}\\
(\vec{A}\times\vec{B})\times(\vec{C}\times\vec{D}) &=& (\vec{A}\times\vec{B}\circ\vec{D})\vec{C} - (\vec{A}\times\vec{B}\circ\vec{C})\vec{D} = [\vec{C},\vec{D},\vec{A}]\vec{B} - [\vec{C},\vec{D},\vec{B}]\vec{A} \quad \mbox{Vector Quadruple Product}\\
\vec{A}\times[\vec{B}\times(\vec{C}\times\vec{D})] &=& (\vec{A}\times\vec{C})(\vec{B}\circ\vec{D}) - (\vec{B}\circ\vec{C})(\vec{A}\times\vec{D})
\end{eqnarray*}

$\textbf{Gradient Identities:}$
\begin{eqnarray*}
\nabla( f +  g) &=& \nabla f + \nabla g\\
\nabla(fg) &=&  f\nabla g +  g\nabla f\\
\nabla(\vec{A}\circ\vec{B}) &=& \vec{A}\times(\nabla\times\vec{B}) + \vec{B}
\times(\nabla\times\vec{A}) + (\vec{A}\circ\nabla)\vec{B} + (\vec{B}\circ\nabla)\vec{A}\\
\nabla(\vec{A}\times\vec{B}) &=& (\nabla\vec{A})\times\vec{B} - (\nabla\vec{B})\times\vec{A} \qquad \mbox{gradient of vector??}\\
\nabla(f\vec{A}) &=& (\nabla f)\vec{A} + f(\nabla\vec{A}) \qquad \mbox{gradient of vector??}
\end{eqnarray*}

$\textbf{Divergence Identities:}$
\begin{eqnarray*}
\nabla\circ(\vec{A} + \vec{B}) &=& \nabla\circ\vec{A} + \nabla\circ\vec{B}\\
\nabla\circ( f\vec{A}) &=& f(\nabla\circ\vec{A}) + \vec{A}\circ(\nabla f)\\
\nabla\circ(\vec{A}\times\vec{B}) &=& \vec{B}\circ(\nabla\times\vec{A}) - \vec{A}\circ(\nabla\times\vec{B})\\
\nabla\circ(\vec{A}\vec{B}) &=& (\nabla\circ\vec{A})\vec{B} + \vec{A}\circ(\nabla\vec{B}) = (\nabla\circ\vec{A})\vec{B} + (\vec{A}\circ\nabla)\vec{B}
\end{eqnarray*}

$\textbf{Curl Identities:}$
\begin{eqnarray*}
\nabla\times(\vec{A} + \vec{B}) &=& \nabla\times\vec{A} + \nabla\times\vec{B}\\
\nabla\times( f\vec{A}) &=& f(\nabla\times\vec{A}) - \vec{A}\times(\nabla f)\\
\nabla\times(\vec{A}\times\vec{B}) &=& (\nabla\circ\vec{B})\vec{A} - (\vec{A}\circ\nabla)\vec{B} + (\vec{B}\circ\nabla)\vec{A} - (\nabla\circ\vec{A})\vec{B}\\
\nabla\times(\vec{A}\vec{B}) &=& (\nabla\circ\vec{A})\vec{B} - \vec{A}\times(\nabla\vec{B})
\end{eqnarray*}

$\textbf{Laplacian Identities:}$
\begin{eqnarray*}
{\nabla}^2(fg) &=& g{\nabla}^2 f + 2\nabla f\circ\nabla g + f{\nabla}^2 g\\
{\nabla}^2(f\vec{A}) &=& = f{\nabla}^2\vec{A} + \vec{A}{\nabla}^2 f + 2(\nabla f\circ\nabla)\vec{A}
\end{eqnarray*}

$\textbf{Mixed Identities:}$
\begin{eqnarray*}
\nabla\circ(\nabla\times\vec{A}) &=& 0\\
\nabla\times\nabla f &=& 0\\
\nabla\times(f\nabla g) &=& \nabla f\times\nabla g\\
\nabla\circ f\nabla g &=& f{\nabla}^2 g + \nabla f\circ\nabla g\\
\nabla\times(\nabla\times\vec{A}) &=& \nabla(\nabla\circ\vec{A}) - {\nabla}^2\vec{A} \quad \mbox{only true for rectangular coordinates}
\end{eqnarray*}

$\textbf{Differential Identities:}$
\begin{eqnarray*}
\frac{d}{dt}( f\vec{A}) &=&  f\frac{d\vec{A}}{dt} + \frac{d f}{dt}\vec{A}\\
\frac{d}{dt}(\vec{A}\circ\vec{B}) &=& \vec{A}\circ\frac{d\vec{B}}{dt} + \vec{B}\circ\frac{d\vec{A}}{dt}\\
\frac{d}{dt}(\vec{A}\times\vec{B}) &=& \vec{A}\times\frac{d\vec{B}}{dt} + \vec{B}\times\frac{d\vec{A}}{dt}
\end{eqnarray*}

$\textbf{Integral Identities:}$

Gauss'/Divergence Thm:  

\indent \indent Standard Form:

\begin{eqnarray*}
\oiint_{\mathcal{S}}\vec{F}\circ d\vec{A} &=& \iiint_{\mathcal{V}}(\nabla\cdot\vec{F})\;dV \quad \mathrm{(\mathcal{S}\,encloses\mathcal{V})}
\end{eqnarray*}

\indent \indent Variants:

\begin{eqnarray*}
\oiint_{\mathcal{S}}\vec{F}\circ\vec{n}\;dS &=& \iiint_{\mathcal{V}}(\nabla\cdot\vec{F})\;dV \quad \mathrm{(\mathcal{S}\,encloses\mathcal{V})} \\
\end{eqnarray*}

Stokes' Thm: 

\indent \indent Standard Form:

\begin{eqnarray*}
\oint_{\mathcal{C}}\vec{F}\circ d\vec{l} &=& \iint_{\mathcal{S}}(\nabla\times\vec{F})\circ\vec{n}\;dS \quad \mathrm{(\mathcal{S}\,bounded\,by\,\mathcal{C})}
\end{eqnarray*}

\indent \indent Variants:

\begin{eqnarray*}
\oint_{\mathcal{C}}\vec{F}\circ d\vec{l} &=& \iint_{\mathcal{S}}(\nabla\times\vec{F})\circ d\vec{A} \quad \mathrm{(\mathcal{S}\,bounded\,by\,\mathcal{C})} \\
\oint_{\mathcal{C}}\vec{F}\circ\vec{t}\;ds &=& \iint_{\mathcal{S}}(\nabla\times\vec{F})\circ\vec{n}\;dS \quad \mathrm{(\mathcal{S}\,bounded\,by\,\mathcal{C})} \\
\oint_{\mathcal{C}}\vec{F}\circ\vec{t}\;ds &=& \iint_{\mathcal{S}}(\nabla\times\vec{F})\circ d\vec{A} \quad \mathrm{(\mathcal{S}\,bounded\,by\,\mathcal{C})} \\
\end{eqnarray*}

Gradient theorem: 

\indent \indent Standard Form:

\begin{eqnarray*}
\int_{\mathcal{C}}(\nabla f)\circ d\vec{l} &=& f(r_2) - f(r_1) \quad \mathrm{(\mathcal{C}\,goes\,from\,r_1\,to\,r_2)}
\end{eqnarray*}

$\textbf{Integration by Parts for Vectors}$
\begin{eqnarray*}
\iiint_{\mathcal{V}} f(\nabla\circ\vec{A})dV &=& \iint_{\mathcal{S}}f\vec{A}\circ d\vec{a} - \iiint_{\mathcal{V}} \vec{A}\circ(\nabla f)dV \\
\end{eqnarray*}

$\textbf{Integral Form of Maxwell's Equations:}$
\begin{eqnarray*}
\oiint_{\mathcal{S}}\vec{D}\circ\hat{n}\;d^2A &=& Q_{free}^{enc} \\
\oiint_{\mathcal{S}}\vec{B}\circ\hat{n}\;d^2A &=& 0 \\
\oint_{\mathcal{C}}\vec{E}\circ d\vec{l} &=& -\frac{d{\Phi_B}}{dt} \\
\oint_{\mathcal{C}}\vec{H}\circ d\vec{l} &=& I_{free}^{enc}+ \frac{d{\Phi_D}}{dt} \\
\Phi_B &=& \iint_{\mathcal{S}}\vec{B}\circ\hat{n}\;d^2A \\
\Phi_D &=& \iint_{\mathcal{S}}\vec{D}\circ\hat{n}\;d^2A \\
\end{eqnarray*}

$\textbf{Differential Form of Maxwell's Equations:}$
\begin{eqnarray*}
\nabla\circ\vec{D} = \rho_{free} \\
\nabla\circ\vec{B} = 0 \\
\nabla\times\vec{E} = -\frac{\partial\vec{B}}{\partial t} \\
\nabla\times\vec{H} = \vec{J}_{free} + \frac{\partial\vec{D}}{\partial t} \\
\end{eqnarray*}

$\textbf{Complex Differential Form of Maxwell's Equations:}$
\begin{eqnarray*}
\nabla\circ\vec{M} = \frac{i\rho}{\epsilon_o} \\
\nabla\times\vec{M} = c\mu_0\vec{J} - \frac{i}{c}\frac{\partial\vec{M}}{\partial t} \\
\text{   where   } \vec{M} \equiv c\vec{B} + i\vec{E} \\ 
\end{eqnarray*}

$\textbf{EM Equations:}$
\begin{eqnarray*}
{\nabla}^2\vec{A} -\frac{1}{c^2}\frac{{\partial}^2 \vec{A}}{\partial t^2} &=& -{\mu}_0\vec{J} \qquad \mbox{Vector Poisson's Equation}\\
{\nabla}^2\vec{V} -\frac{1}{c^2}\frac{{\partial}^2 V}{\partial t^2} &=& \frac{-\rho}{\epsilon} \qquad \mbox{Scalar Poisson's Equation}\\
\vec{A}(\vec{r}) &=& \frac{{\mu}_0}{4\pi} \int \frac{\vec{J}(\vec{r'})}{|\vec{r}-\vec{r'}|} d^3\vec{r'} \\
\phi(\vec{r}) &=& \frac{1}{4\pi{\epsilon}_0} \int \frac{\rho(\vec{r'})}{|\vec{r}-\vec{r'}|} d^3\vec{r'} \\
\end{eqnarray*}
